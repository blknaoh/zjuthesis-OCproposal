% Set default caption font type
\DeclareCaptionFont{figurefont}{\zihao{-5}\heiti}
\DeclareCaptionFont{tablefont}{\zihao{-5}\songti}

\captionsetup[figure]{font=figurefont}
\captionsetup[table]{font=tablefont}

% `position` doesn't actually set the caption position,
% It only determines weather the `skip` inserted above/below `caption`
\captionsetup{position=bottom}
